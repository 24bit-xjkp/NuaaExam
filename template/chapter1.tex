% 便于编写时查看排版效果
\ifdefined\Solution
\else
    \def\Solution{inline}
\fi

% 通用设置
\examsetup
{
    fillin/type = line,
    fillin/no-answer-type = none,
    solution/show-qed = false,
    paren/type = none
}

% 内联答案则选择填空显示答案
\ifdefstring{\Solution}{inline}
{
    \examsetup
    {
        question/show-answer = true,
        paren/show-answer = true
    }
}
{
    \examsetup
    {
        question/show-answer = false,
        paren/show-answer = false
    }
}
% 内联答案则原位显示解答
\ifdefstring{\Solution}{inline}
{\examsetup{solution/show-solution = show-stay}}
{
    % 后置答案则将解答移动到最后
    \ifdefstring{\Solution}{backend}
    {\examsetup{solution/show-solution = show-move}}
    % 隐藏答案和混合模式则不显示任何解答
    {\examsetup{solution/show-solution = hide}}
}

% 生成混合模式的试题章节
\ifdefstring{\Solution}{mixed}
{
    \tableofcontents
    \chapter{试题}
}{}
